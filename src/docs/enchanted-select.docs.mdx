import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedSelectStories from '../stories/enchanted-select.stories';
import { INPUT_SELECT_PARTS } from '../types/cssClassEnums';

export const INPUT_SELECT_PARTS_METADATA = new Map([
  [INPUT_SELECT_PARTS.DIV,'Styles wrapper around the select field.'],
  [INPUT_SELECT_PARTS.DIV_LABEL,'Styles label row above the filed.'],
  [INPUT_SELECT_PARTS.LABEL,'Styles visible label text.'],
  [INPUT_SELECT_PARTS.LABEL_DISABLED,'Styles disabled label.'],
  [INPUT_SELECT_PARTS.REMOVE_LABEL,'Styles removable label control.'],
  [INPUT_SELECT_PARTS.REMOVE_LABEL_DISABLED,'Styles disabled removable label.'],
]);

export const CSSPARTS = Object.values(INPUT_SELECT_PARTS).map((part) => ({
    name: part,
    description: INPUT_SELECT_PARTS_METADATA.get(part) ?? ''
}));
