import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedPanelStories from '../stories/enchanted-panel.stories';
import { PANEL_PARTS, PANEL_SLOTS } from '../types/cssClassEnums';

export const PANEL_PARTS_METADATA = new Map([
  [PANEL_PARTS.PANEL_CONTAINER, 'Styles main panel wrapper.'],
  [PANEL_PARTS.PANEL_HEADER, 'Styles header section with title.'],
  [PANEL_PARTS.PANEL_TITLE, 'Styles title text element.'],
  [PANEL_PARTS.PANEL_CONTENT, 'Styles main content area.'],
  [PANEL_PARTS.PANEL_CLOSE_BUTTON, 'Styles close/dismiss button.'],

]);

export const CSSPARTS = Object.values(PANEL_PARTS).map((part) => ({
    name: part,
    description: PANEL_PARTS_METADATA.get(part) ?? ''
}));

export const PANEL_SLOTS_METADATA = new Map([
  [PANEL_SLOTS.CENTER_TITLE_CONTENT, 'Optional. Used to provide custom content in the center of the panel header, such as actions, indicators, or additional controls. If omitted, only the default header title and close button are displayed.' ],
  [PANEL_SLOTS.CONTENT, 'Required. Used to provide the main content displayed inside the panel. If omitted, the panel is rendered without any body content' ],
]);

export const slots = Object.values(PANEL_SLOTS).map((slot) => ({
  name: slot,
  description: PANEL_SLOTS_METADATA.get(slot) ?? ''
}));
