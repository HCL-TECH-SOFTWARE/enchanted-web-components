export const TOOLTIP_PARTS_METADATA = new Map([
  [TOOLTIP_PARTS.TOOLTIP_ROOT,'Styles root container for the tooltip wrapper.'],
  [TOOLTIP_PARTS.TOOLTIP_TEXT,'Styles tooltip bubble content.'],

    name: part,
    description: TOOLTIP_PARTS_METADATA.get(part) ?? ''
}));

## Slots

| **Slot** | **Description** |
|----------|-----------------|
| `tooltip` | **Optional.** Used to provide custom tooltip content. If omitted, the component displays the value of the `tooltiptext` property.   |
| `target` | **Required.** Used to provide the element that displays the tooltip when hovered or focused. If omitted, the tooltip has no target and cannot be shown. |
