export const CIRCULAR_PROGRESS_PART_METADATA = new Map([

[CIRCULAR_PROGRESS_PARTS.SVG , 'SVG element that renders the progress ring.'],
[CIRCULAR_PROGRESS_PARTS.TRACK , 'Track circle behind the progress ring.'],
[CIRCULAR_PROGRESS_PARTS.CIRCLE , 'Animated progress circle.'],
[CIRCULAR_PROGRESS_PARTS.CIRCLE_DISABLE_SHRINK , 'Progress circle without shrink.'],
[CIRCULAR_PROGRESS_PARTS.LABEL , 'Label shown under the spinner.'],
[CIRCULAR_PROGRESS_PARTS.SPINNER , 'Spinner container around the SVG.'],
]);

export const CSSPARTS = Object.values(CIRCULAR_PROGRESS_PARTS).map((part) => ({
    name: part,
    description: CIRCULAR_PROGRESS_PART_METADATA.get(part) ?? ''
}));
