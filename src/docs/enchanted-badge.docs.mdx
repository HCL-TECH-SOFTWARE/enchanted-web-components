import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedBadgeStories from '../stories/enchanted-badge.stories';
import { EnchantedBadgeParts } from '../types/cssClassEnums';

export const EnchantedBadgeParts_METADATA = new Map([
   [EnchantedBadgeParts.BADGE_TEXT, 'text badge container.'],
   [EnchantedBadgeParts.BADGE_DOT, 'dot badge container.'],
   [EnchantedBadgeParts.BADGE_TEXT_RTL, 'RTL text badge container.'],
]);

export const CSSPARTS = Object.values(EnchantedBadgeParts).map((part) => ({
    name: part,
    description: EnchantedBadgeParts_METADATA.get(part) ?? ''
}));
