export const TOGGLE_BUTTON_SLOTS_METADATA = new Map([
  [TOGGLE_BUTTON_SLOTS.ICON,  'Required. Used to provide the icon displayed inside the toggle button. If omitted, the component is rendered without an icon, which may result in an incomplete or unusable UI.' ],

]);

export const CSSPARTS = Object.values(TOGGLE_BUTTON_PARTS).map((part) => ({
    name: part,
    description: TOGGLE_BUTTON_PARTS_METADATA.get(part) ?? ''
}));

export const slots = Object.values(TOGGLE_BUTTON_SLOTS).map((slot) => ({
  name: slot,
  description: TOGGLE_BUTTON_SLOTS_METADATA.get(slot) ?? ''
}));
