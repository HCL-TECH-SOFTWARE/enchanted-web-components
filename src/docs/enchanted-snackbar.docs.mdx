import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedSnackbarStories from '../stories/enchanted-snackbar.stories';
import { SNACKBAR_PARTS, SNACKBAR_SLOTS } from '../types/cssClassEnums';

export const SNACKBAR_PARTS_METADATA = new Map([
  [SNACKBAR_PARTS.SNACKBAR_CONTAINER,'Styles root snackbar container.'],
  [SNACKBAR_PARTS.SNACKBAR_MESSAGE,'Styles message text container.'],
  [SNACKBAR_PARTS.SNACKBAR_BUTTON_CONTAINER,'Styles wrapper for snackbar buttons.'],
  [SNACKBAR_PARTS.SNACKBAR_BUTTONS,'Styles slot area for the snackbar message.'],
  [SNACKBAR_PARTS.SNACKBAR_ICON,'Styles icon shown for the snackbar message.'],
  [SNACKBAR_PARTS.SNACKBAR_PROGRESS,'Styles progress spinner container for progress snackbar.'],
  [SNACKBAR_PARTS.SNACKBAR_ICON_CONTAINER,'Styles container around the snackbar icon.'],
]);

export const CSSPARTS = Object.values(SNACKBAR_PARTS).map((part) => ({
    name: part,
    description: SNACKBAR_PARTS_METADATA.get(part) ?? ''
}));

export const SNACKBAR_SLOTS_METADATA = new Map([
  [SNACKBAR_SLOTS.SNACKBAR_BUTTONS,  'Optional. Used to provide one or more action buttons displayed in the snackbar. If omitted, the snackbar is rendered without any action buttons.' ],
]);

export const slots = Object.values(SNACKBAR_SLOTS).map((slot) => ({
  name: slot,
  description: SNACKBAR_SLOTS_METADATA.get(slot) ?? ''
}));

<Meta of={EnchantedSnackbarStories} />

<Title />

The `enchanted-snackbar` displays brief temporary notifications at the bottom of the screen with five visual types (info, success, warning, error, progress). Supports HTML content and optional action buttons for user interactions.
