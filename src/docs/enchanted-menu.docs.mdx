import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedMenuStories from '../stories/enchanted-menu.stories';
import { MENU_PARTS, MENU_ITEM_PARTS  } from '../types/cssClassEnums';

export const MENU_PARTS_METADATA = new Map([
  [MENU_PARTS.MENU_ROOT, 'Root container of the overlay menu.'],
  [MENU_PARTS.BACKDROP, 'Backdrop behind the menu.'],
  [MENU_PARTS.PAPER_ROOT, 'Menu paper/surface container.'],
  [MENU_ITEM_PARTS.TEXT_ROOT, 'Wrapper around the menu item text.'],
  [MENU_ITEM_PARTS.TEXT, 'The visible menu item label.'],
]);

export const CSSPARTS = [...Object.values(MENU_PARTS), ...Object.values(MENU_ITEM_PARTS)].map((part) => ({
    name: part,
    description: MENU_PARTS_METADATA.get(part) ?? ''
}));
