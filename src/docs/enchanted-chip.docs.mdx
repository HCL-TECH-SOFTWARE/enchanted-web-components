import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedChipStories from '../stories/enchanted-chip.stories';
import { CHIP_PARTS, CHIP_SLOTS } from '../types/cssClassEnums';

export const CHIP_PART_METADATA = new Map([
  [CHIP_PARTS.CHIP_DIV, 'Main chip container.'],
  [CHIP_PARTS.CHIP_NAME, 'Chip label text.'],
  [CHIP_PARTS.CHIP_AVATAR, 'Avatar area inside the chip.'],
  [CHIP_PARTS.CHIP_COUNT, 'Count badge shown within the chip.'],
  [CHIP_PARTS.CHIP_COUNT_RTL, 'RTL count badge styling.'],
  [CHIP_PARTS.CHIP_DIV_DISABLED, 'Disabled chip appearance.'],
]);

export const CSSPARTS = Object.values(CHIP_PARTS).map((part) => ({
    name: part,
    description: CHIP_PART_METADATA.get(part) ?? ''
}));

export const CHIP_SLOTS_METADATA = new Map([
  [CHIP_SLOTS.CLEAR_ICON,  'Optional. Used to provide a custom clear or remove icon displayed at the end of the chip when `clearIcon` is enabled. If omitted, no clear icon is rendered.' ],
]);

export const slots = Object.values(CHIP_SLOTS).map((slot) => ({
  name: slot,
  description: CHIP_SLOTS_METADATA.get(slot) ?? ''
}));

<Meta of={EnchantedChipStories} />

<Title />

The `enchanted-chip` displays compact elements with optional avatar, count badge, and clear icon for dismissal. Use chips for tags, filters, or selected items with full RTL support and disabled state styling.
