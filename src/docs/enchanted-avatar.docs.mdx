  [AVATAR_PARTS.AVATAR_ICON_TEMPLATE_CIRCULAR , 'circular avatar content rendered from a template.'],
  [AVATAR_PARTS.AVATAR_ICON_TEMPLATE_ROUNDED , 'rounded avatar content rendered from a template.'],
