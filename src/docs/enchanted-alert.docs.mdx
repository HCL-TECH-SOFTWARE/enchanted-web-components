import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedAlertStories from '../stories/enchanted-alert.stories';
import { ALERT } from '../types/cssClassEnums';

export const ALERT_PART_METADATA = new Map([
  [ALERT.ALERT_DIV_ROOT, 'Styles the alert root container.'],
  [ALERT.ALERT_CONTAINED_INFO, 'Styles the contained alert in the info state.'],
  [ALERT.ALERT_CONTAINED_ERROR, 'Styles the contained alert in the error state.'],
  [ALERT.ALERT_CONTAINED_SUCCESS, 'Styles the contained alert in the success state.'],
  [ALERT.ALERT_CONTAINED_WARNING, 'Styles the contained alert in the warning state.'],
  [ALERT.ALERT_OUTLINED_INFO, 'Styles the outlined alert in the info state.'],
  [ALERT.ALERT_OUTLINED_ERROR, 'Styles the outlined alert in the error state.'],
  [ALERT.ALERT_OUTLINED_SUCCESS, 'Styles the outlined alert in the success state.'],
  [ALERT.ALERT_OUTLINED_WARNING, 'Styles the outlined alert in the warning state.'],
  [ALERT.ALERT_SVG_SUCCESS, 'Styles the alert icon in the success state.'],
  [ALERT.ALERT_SVG_WARNING, 'Styles the alert icon in the warning state.'],
  [ALERT.ALERT_SVG_INFO, 'Styles the alert icon in the info state.'],
  [ALERT.ALERT_SVG_ERROR, 'Styles the alert icon in the error state.'],
  [ALERT.ALERT_CONTENT, 'Styles the alert content area.'],
  [ALERT.ALERT_TITLE, 'Styles the alert title text.'],
]);

export const CSSPARTS = Object.values(ALERT).map((part) => ({
    name: part,
    description: ALERT_PART_METADATA.get(part) ?? ''
}));

<Meta of={EnchantedAlertStories} />

<Title />

The `enchanted-alert` displays important messages with multiple severity levels (info, success, warning, error) and visual variants (contained, outlined). Each severity has a corresponding icon and color scheme for visual feedback.
