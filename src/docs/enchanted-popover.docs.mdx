export const POPOVER_PARTS_METADATA = new Map([
  [POPOVER_PARTS.POPOVER_WRAPPER,'Styles Wrapper around the popover content.'],
  [POPOVER_PARTS.POPOVER_ARROW,'Styles arrow element for the popover.'],
  [POPOVER_PARTS.POPOVER_CONTAINER,'Styles popover content container.'],
  [POPOVER_PARTS.POPOVER_CONTAINER_RTL,'Styles RTL popover content container.'],
  [POPOVER_PARTS.POPOVER_CONTENT,'Styles content area.'],
  [POPOVER_PARTS.POPOVER_LABEL,'Styles label content slot.'],
  [POPOVER_PARTS.POPOVER_TEXT,'Styles text content slot.'],
  [POPOVER_PARTS.POPOVER_CLOSE_ICON,'Styles close icon button.'],
  [POPOVER_PARTS.POPOVER_CLOSE_ICON_RTL,'Styles RTL close icon button.'],
  [POPOVER_PARTS.POPOVER_TARGET,'Styles target element that anchors the popover.'],
]);

export const CSSPARTS = Object.values(POPOVER_PARTS).map((part) => ({
    name: part,
    description: POPOVER_PARTS_METADATA.get(part) ?? ''
}));

<Title />

The `enchanted-popover` displays contextual content in a floating panel anchored to a target element. Supports 13 arrow positions, dark/light themes, optional close buttons, and full RTL support for flexible positioning.
