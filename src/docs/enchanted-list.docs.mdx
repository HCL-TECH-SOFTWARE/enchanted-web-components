import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedListStories from '../stories/enchanted-list.stories';
import { LIST_PARTS, LIST_ITEM_PARTS } from '../types/cssClassEnums';

export const LIST_PARTS_METADATA = new Map([
  [LIST_PARTS.UNORDERED_LIST, 'Unordered list container.'],
  [LIST_ITEM_PARTS.LIST_ITEM, 'Standard list item.'],
  [LIST_ITEM_PARTS.LIST_ITEM_SELECTED, 'Selected list item state.'],
  [LIST_ITEM_PARTS.MENU_ITEM, 'Menu-specific list item.'],
]);

export const CSSPARTS = [...Object.values(LIST_PARTS),...Object.values(LIST_ITEM_PARTS)].map((part) => ({
    name: part,
    description: LIST_PARTS_METADATA.get(part) ?? ''
}));

<Meta of={EnchantedListStories} />

<Title />
