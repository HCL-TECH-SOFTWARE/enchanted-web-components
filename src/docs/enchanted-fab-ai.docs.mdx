import { Meta, Title, Canvas, Controls, Stories } from '@storybook/addon-docs/blocks';
import * as EnchantedFabAiStories from '../stories/enchanted-fab-ai.stories';
import { FAB_PARTS, FAB_AI_SLOTS } from '../types/cssClassEnums';

export const FAB_PART_METADATA = new Map([
  [FAB_PARTS.FAB, 'Main floating action button part.'],
  [FAB_PARTS.FAB_RTL, 'RTL fab styling.'],
  [FAB_PARTS.LABEL, 'Label text inside an extended fab.'],
  [FAB_PARTS.LABEL_RTL, 'RTL label styling.'],
  [FAB_PARTS.ICON, 'Icon container inside the fab.'],
]);

export const CSSPARTS = Object.values(FAB_PARTS).map((part) => ({
    name: part,
    description: FAB_PART_METADATA.get(part) ?? ''
}));

export const FAB_AI_SLOTS_METADATA = new Map([
  [FAB_AI_SLOTS.BADGE,  'Optional. Used to provide badge content displayed alongside the Floating Action Button when `badge` is enabled. If omitted, no badge is rendered.' ],
]);

export const slots = Object.values(FAB_AI_SLOTS).map((slot) => ({
  name: slot,
  description: FAB_AI_SLOTS_METADATA.get(slot) ?? ''
}));
